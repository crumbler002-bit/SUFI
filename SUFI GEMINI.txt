TypeScript
"use client";

import { useRef, useState } from "react";
import { motion } from "framer-motion";
import { cn } from "@/lib/utils"; // Standard Tailwind merge utility
import { sufiSpring } from "@/lib/anim-config";

interface SpotlightCardProps {
  children: React.ReactNode;
  className?: string;
  onClick?: () => void;
}

export function SpotlightCard({ children, className, onClick }: SpotlightCardProps) {
  const divRef = useRef<HTMLDivElement>(null);
  const [position, setPosition] = useState({ x: 0, y: 0 });
  const [opacity, setOpacity] = useState(0);

  // We track the mouse relative to the card's bounding box
  const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {
    if (!divRef.current) return;
    const rect = divRef.current.getBoundingClientRect();
    setPosition({ x: e.clientX - rect.left, y: e.clientY - rect.top });
  };

  return (
    <motion.div
      ref={divRef}
      onMouseMove={handleMouseMove}
      onMouseEnter={() => setOpacity(1)}
      onMouseLeave={() => setOpacity(0)}
      onClick={onClick}
      whileHover={{ scale: 0.995 }} // Subtle press effect
      transition={sufiSpring}
      className={cn(
        "relative overflow-hidden rounded-2xl bg-neutral-900 border border-white/5 cursor-pointer shadow-2xl shadow-black/50 transition-colors hover:border-white/10",
        className
      )}
    >
      {/* THE SPOTLIGHT ENGINE 
        This div is invisible until hovered. It paints a gradient centered exactly on the cursor.
      */}
      <div
        className="pointer-events-none absolute -inset-px transition-opacity duration-300 z-0"
        style={{
          opacity,
          background: `radial-gradient(600px circle at ${position.x}px ${position.y}px, rgba(255,255,255,0.06), transparent 40%)`,
        }}
      />
      
      {/* We duplicate the spotlight purely for the BORDER glow. 
        This is a classic WebGL/CSS trick to make borders light up selectively.
      */}
      <div
        className="pointer-events-none absolute inset-0 z-0 rounded-2xl opacity-0 transition-opacity duration-300 mix-blend-overlay ring-1 ring-white/20"
        style={{
          opacity,
          background: `radial-gradient(400px circle at ${position.x}px ${position.y}px, rgba(255,255,255,0.4), transparent 40%)`,
        }}
      />

      {/* THE CONTENT PAYLOAD */}
      <div className="relative z-10 p-6 flex flex-col h-full">
        {children}
      </div>
    </motion.div>
  );
}

import { create } from "zustand";

// 1. Strict typing for the exact phases of the SUFI decision engine
export type EngineState = 
  | "idle"        // Waiting for user intent
  | "analyzing"   // pgvector is running, show text scramble
  | "revealing"   // pgvector finished, trigger GSAP cinematic reveal
  | "resolved";   // Animation done, UI is fully interactive

interface SufiState {
  engineState: EngineState;
  activeQuery: string;
  setEngineState: (state: EngineState) => void;
  setActiveQuery: (query: string) => void;
  
  // A transient function to trigger the full flow
  initiateDecisionEngine: (query: string) => Promise<void>;
}

export const useSufiStore = create<SufiState>((set) => ({
  engineState: "idle",
  activeQuery: "",
  
  setEngineState: (state) => set({ engineState: state }),
  setActiveQuery: (query) => set({ activeQuery: query }),

  initiateDecisionEngine: async (query) => {
    set({ activeQuery: query, engineState: "analyzing" });
    
    // Simulate the API call to the FastAPI pgvector endpoint
    // In production, this is where you await your fetch/axios call
    await new Promise((resolve) => setTimeout(resolve, 1500));
    
    // API returns the best match. We shift to the GSAP reveal phase.
    set({ engineState: "revealing" });
    
    // Allow GSAP time to run its timeline, then unlock the UI
    setTimeout(() => {
      set({ engineState: "resolved" });
    }, 2500); // Matches the GSAP timeline duration
  },
}));

Elite SaaS applications don't expose their API fetching logic or database shapes to the client browser. We will use Next.js Server Actions as a secure middleware layer. Create actions/menu-actions.ts:

TypeScript
"use server";

import { revalidatePath } from "next/cache";
import { z } from "zod";

// 1. Strict input validation - never trust the client
const UpdateMenuSchema = z.object({
  itemId: z.string(),
  price: z.number().positive(),
  available: z.boolean(),
});

// 2. The Server Action called directly from our Liquid Menu Editor
export async function updateMenuItem(formData: z.infer<typeof UpdateMenuSchema>) {
  // Validate the payload before it ever touches FastAPI
  const validatedFields = UpdateMenuSchema.safeParse(formData);
  
  if (!validatedFields.success) {
    return { error: "Invalid data payload." };
  }

  const { itemId, price, available } = validatedFields.data;

  try {
    // 3. Secure Server-to-Server fetch to your FastAPI backend
    // No CORS issues, no exposed API keys.
    const response = await fetch(`${process.env.FASTAPI_INTERNAL_URL}/owner/menu/${itemId}`, {
      method: "PUT",
      headers: {
        "Content-Type": "application/json",
        // Inject secure auth token from Next.js cookies
        "Authorization": `Bearer ${process.env.INTERNAL_SERVICE_KEY}`, 
      },
      body: JSON.stringify({ price, available }),
    });

    if (!response.ok) throw new Error("FastAPI rejected the update");

    // 4. Instantly purge the Next.js cache so the UI updates without a reload
    revalidatePath("/dashboard/menu");
    
    return { success: true };
  } catch (error) {
    console.error("Failed to update menu item:", error);
    return { error: "System failure. Please try again." };
  }
}

class MenuItem(Base):

2. The Alembic Migration (Injecting the Hardware Acceleration)
When you run alembic revision --autogenerate, it creates a migration file. However, standard auto-generation doesn't know how to optimize vector databases. We have to manually inject an HNSW (Hierarchical Navigable Small World) index.

This index is the secret to making your AI semantic search blistering fast, even if you have 100,000 restaurants in the database.

Modify your Alembic migration file (versions/xxxx_initial_schema.py):

Python
"""initial schema with pgvector

Revision ID: 8a9b2c3d4e5f
Create Date: 2026-03-27 10:00:00.000000
"""
from alembic import op
import sqlalchemy as sa
from pgvector.sqlalchemy import Vector

def upgrade():
    # 1. CRITICAL: Enable the pgvector extension on the database
    op.execute('CREATE EXTENSION IF NOT EXISTS vector')

    # 2. Create the standard tables
    op.create_table('users',
        sa.Column('id', sa.String(), nullable=False),
        sa.Column('role', sa.String(), nullable=False), # 'diner', 'owner', 'admin'
        sa.PrimaryKeyConstraint('id')
    )

    op.create_table('restaurants',
        sa.Column('id', sa.String(), nullable=False),
        sa.Column('owner_id', sa.String(), nullable=False),
        sa.Column('name', sa.String(), nullable=False),
        sa.Column('lat', sa.Float(), nullable=False),
        sa.Column('lng', sa.Float(), nullable=False),
        sa.Column('embedding', Vector(1536)),
        sa.Column('rating', sa.Float(), nullable=True),
        sa.Column('trending_score', sa.Float(), nullable=True),
        sa.Column('tier_boost', sa.Float(), nullable=True),
        sa.ForeignKeyConstraint(['owner_id'], ['users.id'], ),
        sa.PrimaryKeyConstraint('id')
    )

    # 3. CRITICAL: Build the HNSW Vector Index for blazing fast Cosine Similarity
    # This allows the database to skip scanning every row when searching for intent
    op.execute("""
        CREATE INDEX ON restaurants 
        USING hnsw (embedding vector_cosine_ops)
        WITH (m = 16, ef_construction = 64);
    """)

def downgrade():
    op.drop_table('restaurants')
    op.drop_table('users')
    op.execute('DROP EXTENSION IF EXISTS vector')

from passlib.context import CryptContext
from jose import jwt

# 1. Initialize the Bcrypt hasher
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

SECRET_KEY = "your-super-secret-jwt-key"
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 60 * 24 * 7 # 7-day session

# 2. Core Security Utilities
def verify_password(plain_password: str, hashed_password: str) -> bool:
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password: str) -> str:
    return pwd_context.hash(password)

def create_access_token(data: dict, expires_delta: timedelta | None = None) -> str:
    to_encode = data.copy()
    expire = datetime.utcnow() + (expires_delta if expires_delta else timedelta(minutes=15))
    to_encode.update({"exp": expire})
    
    # Sign the JWT
    encoded_jwt = jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)
    return encoded_jwt

import { auth } from "@/auth"; // Our NextAuth setup

  region: process.env.AWS_REGION!,
  credentials: {
    accessKeyId: process.env.AWS_ACCESS_KEY_ID!,
    secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY!,
  },
});

export async function getPresignedUploadUrl(fileType: string, fileSize: number) {
  // 1. Security Check: Only logged-in owners/admins can upload
  const session = await auth();
  if (!session || session.user.role === "diner") {
    return { error: "Unauthorized upload attempt." };
  }

  // 2. Validation Check: Prevent massive files or non-images
  const MAX_FILE_SIZE = 5 * 1024 * 1024; // 5MB
  if (fileSize > MAX_FILE_SIZE) return { error: "File exceeds 5MB limit." };
  if (!fileType.startsWith("image/")) return { error: "Only images are allowed." };

  // 3. Generate a secure, randomized file name
  const extension = fileType.split("/")[1];
  const key = `restaurants/${session.user.id}/${uuidv4()}.${extension}`;

  // 4. Request the Presigned URL from S3
  const command = new PutObjectCommand({
    Bucket: process.env.AWS_BUCKET_NAME!,
    Key: key,
    ContentType: fileType,
  });

  try {
    // This URL is only valid for 60 seconds
    const signedUrl = await getSignedUrl(s3, command, { expiresIn: 60 });
    
    return { 
      success: true, 
      uploadUrl: signedUrl, 
      finalImageUrl: `https://${process.env.AWS_BUCKET_NAME}.s3.amazonaws.com/${key}` 
    };
  } catch (error) {
    console.error("Presigned URL generation failed:", error);
    return { error: "Failed to initialize upload pipeline." };
  }
}

TypeScript
"use client";

import { MouseEvent } from "react";
import { motion, useMotionTemplate, useMotionValue } from "framer-motion";

export function SpotlightCard({ children, className = "" }: { children: React.ReactNode; className?: string }) {
  // Motion values bypass React state, ensuring zero re-renders during mouse movement
  const mouseX = useMotionValue(0);
  const mouseY = useMotionValue(0);

  function handleMouseMove({ currentTarget, clientX, clientY }: MouseEvent) {
    const { left, top } = currentTarget.getBoundingClientRect();
    mouseX.set(clientX - left);
    mouseY.set(clientY - top);
  }

  return (
    <div
      onMouseMove={handleMouseMove}
      className={`group relative flex w-full flex-col overflow-hidden rounded-2xl border border-white/10 bg-[#0A0A0A] shadow-lg transition-colors hover:border-white/20 ${className}`}
    >
      {/* THE SPOTLIGHT ENGINE */}
      <motion.div
        className="pointer-events-none absolute -inset-px rounded-2xl opacity-0 transition-opacity duration-300 group-hover:opacity-100"
        style={{
          background: useMotionTemplate`
            radial-gradient(
              400px circle at ${mouseX}px ${mouseY}px,
              rgba(255, 255, 255, 0.06),
              transparent 80%
            )
          `,
        }}
      />
      
      {/* THE CONTENT */}
      <div className="relative z-10 flex h-full flex-col">
        {children}
      </div>
    </div>
  );
}

To be a 2026 AI-Native Operating System, the dashboard cannot be passive. The LLM cannot just be a chatbot hidden in a corner; the LLM must be the nervous system of the dashboard itself.

Here is how we engineer the 99.9th percentile AI-Native Dashboard for SUFI OS.

1. The Paradigm Shift: Passive Data vs. Generative Operations (GenOps)
Standard SaaS (Linear/Stripe): Data changes -> WebSocket fires -> Chart line moves. The owner has to look at the chart, interpret the spike, and figure out what to do.

SUFI OS (AI-Native): Data changes -> FastAPI triggers LLM -> LLM analyzes the delta -> LLM streams UI components directly onto the dashboard. ### 2. The Elite Mechanics: How the LLM Changes the OS

We achieve this by giving the LLM the ability to dispatch specific JSON payloads through your WebSocket/SSE stream, which our Framer Motion UI interprets as physical components.

Mechanic A: Autonomous Chart Annotations
When the LiveVelocityChart spikes because 5 tables were booked in 10 minutes, the line doesn't just go up. The LLM detects the anomaly and injects a Context Node directly onto the SVG path.

Visual: A glowing dot appears on the chart's peak. A Framer Motion tooltip expands automatically: "Surge detected: Theater district just let out. 80% of these bookings are parties of 2."

Mechanic B: Predictive "Ghost" UI
The LLM doesn't just read the present; it predicts the next 2 hours.

Visual: On your charts and floorplan, the current reality is rendered in solid, high-contrast colors (white/emerald). The LLM's prediction for the next hour is rendered ahead of the timeline using a dashed SVG stroke and a blurred, 30% opacity "ghost" UI.

Mechanic C: Transient Action Widgets
If the LLM sees the kitchen is getting backed up (based on table turn times), it shouldn't just show a red chart. It should physically render a temporary control widget on the dashboard.

Visual: A widget smoothly expands from the top of the dashboard: "Kitchen load critical. Recommend pausing walk-ins for 30 minutes to protect Quality Signal." with a single [Execute] button. Once clicked, the widget dissolves.

import { useEffect, useState } from "react";
import { motion, AnimatePresence } from "framer-motion";
import { AlertTriangle, Zap, CheckCircle2 } from "lucide-react";

// The physics for when the LLM injects a new component
const llmSpring = { type: "spring", stiffness: 350, damping: 25 };

interface LlmActionPayload {
  id: string;
  type: "insight" | "warning" | "action_required";
  title: string;
  description: string;
  suggestedAction?: {
    label: string;
    apiEndpoint: string;
  };
}

export function AiNervousSystem({ children }: { children: React.ReactNode }) {
  const [activeInjections, setActiveInjections] = useState<LlmActionPayload[]>([]);

  // Simulated WebSocket connection to the FastAPI LLM stream
  useEffect(() => {
    // In production: Connect to ws://api.sufi.com/dashboard/stream
    const triggerSimulatedLlmInsight = setTimeout(() => {
      setActiveInjections(prev => [...prev, {
        id: "evt_123",
        type: "action_required",
        title: "Capacity Warning",
        description: "Velocity spike detected. Waitlist is exceeding 45 minutes. Recommend throttling external discovery.",
        suggestedAction: { label: "Pause Walk-ins (30m)", apiEndpoint: "/api/throttle" }
      }]);
    }, 5000);

    return () => clearTimeout(triggerSimulatedLlmInsight);
  }, []);

  const executeLlmAction = async (id: string, endpoint: string) => {
    // 1. Hit the Server Action
    // await executeFastApiAction(endpoint);
    
    // 2. Dismiss the injection
    setActiveInjections(prev => prev.filter(inj => inj.id !== id));
  };

  return (
    <div className="relative w-full">
      {/* 1. The LLM Injection Layer (Sits at the top of the dashboard) */}

              key={injection.id}
              initial={{ opacity: 0, y: -20, scale: 0.95 }}
              animate={{ opacity: 1, y: 0, scale: 1 }}
              exit={{ opacity: 0, scale: 0.95, filter: "blur(5px)" }}
              transition={llmSpring}
              className={`pointer-events-auto flex items-center justify-between gap-4 rounded-2xl border bg-[#0A0A0A]/95 p-4 shadow-2xl backdrop-blur-xl ${

              }`}
            >
              <div className="flex items-start gap-4">
                <div className={`mt-0.5 flex h-8 w-8 shrink-0 items-center justify-center rounded-full ${

                >

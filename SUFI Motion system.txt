⚡ 2. SearchSystem (MAIN)
SearchSystem.tsx
"use client";

import { useSufiStore } from "@/lib/state/sufiStore";
import SearchInput from "./SearchInput";
import ResultsGrid from "./ResultsGrid";
import BestMatchPanel from "./BestMatchPanel";

export default function SearchSystem() {
  const { results, bestMatchId, explanations, systemStatus } = useSufiStore();

  const bestResult = results.find(r => r.id === bestMatchId);

  return (
    <div className="max-w-5xl mx-auto px-6 py-12 space-y-8">
      
      {/* Search */}
      <SearchInput />

      {/* Thinking state */}
      {systemStatus === "searching" && (
        <div className="text-sm text-white/30 animate-pulse">
          Understanding your intent...
        </div>
      )}

      {/* Best Match (decision moment) */}
      {bestResult && (
        <BestMatchPanel
          data={bestResult}
          explanation={explanations[bestResult.id]}
        />
      )}

      {/* Results */}
      <ResultsGrid results={results} />
    </div>
  );
}
🔍 3. SearchInput
"use client";

import { useState } from "react";
import { useSufiStore } from "@/lib/state/sufiStore";

export default function SearchInput() {
  const [value, setValue] = useState("");
  const { search } = useSufiStore();

  return (
    <input
      value={value}
      onChange={(e) => setValue(e.target.value)}
      onKeyDown={(e) => e.key === "Enter" && search(value)}
      placeholder="Find restaurants..."
      className="w-full bg-white/[0.04] border border-white/10 rounded-2xl px-6 py-4 text-white placeholder:text-white/30 focus:border-primary/40 outline-none"
    />
  );
}

import { create } from "zustand";

export const useRestaurantStore = create((set) => ({
  waitlist: [
    { id: 1, size: 2, wait: 10 },
    { id: 2, size: 4, wait: 20 },
  ],

  tables: [
    { id: 1, capacity: 2, status: "occupied" },
    { id: 2, capacity: 2, status: "free" },
    { id: 3, capacity: 4, status: "free" },
  ],

  suggestions: [],

  seatParty: (partyId) =>
    set((state) => ({
      waitlist: state.waitlist.filter((p) => p.id !== partyId),
    })),

  updateSuggestions: () =>
    set((state) => {
      const suggestions = [];

      state.waitlist.forEach((p) => {
        const table = state.tables.find(
          (t) => t.capacity >= p.size && t.status === "free"
        );

        if (table) {
          suggestions.push({
            text: `Seat party of ${p.size} at Table ${table.id}`,
            partyId: p.id,
          });
        }
      });

      return { suggestions };
    }),
}));

from fastapi import FastAPI, WebSocket

app = FastAPI()

clients = []

@app.websocket("/ws")
async def websocket_endpoint(ws: WebSocket):
    await ws.accept()
    clients.append(ws)

    try:
        while True:
            data = await ws.receive_json()

            # broadcast update to all clients
            for client in clients:
                await client.send_json(data)

    except:
        clients.remove(ws)

import { useEffect } from "react";
import { useRestaurantStore } from "@/lib/state/restaurantStore";

export default function useSocket() {
  const { setWaitlist, setTables } = useRestaurantStore();

  useEffect(() => {
    const ws = new WebSocket("ws://localhost:8000/ws");

    ws.onmessage = (event) => {
      const msg = JSON.parse(event.data);

      if (msg.type === "WAITLIST_UPDATE") {
        setWaitlist(msg.data);
      }

      if (msg.type === "TABLE_UPDATE") {
        setTables(msg.data);
      }
    };

    return () => ws.close();
  }, []);
}

def assign_tables(waitlist, tables):
    assignments = []

    for party in waitlist:
        best_table = None
        best_score = 0

        for table in tables:
            if table["status"] == "free" and table["capacity"] >= party["size"]:
                score = compute_fit(party, table)

                if score > best_score:

                    best_table = table

🌐 4. WebSocket Manager
/ws/manager.py
class WSManager:
    def __init__(self):
        self.clients = []

    async def connect(self, ws):
        await ws.accept()
        self.clients.append(ws)

    def disconnect(self, ws):
        self.clients.remove(ws)

    async def broadcast(self, message):
        for client in self.clients:
            await client.send_json(message)

@app.websocket("/ws")
async def websocket_endpoint(ws: WebSocket):

    try:
        while True:
            event = await ws.receive_json()

            # update state
            if event["type"] == "WAITLIST_UPDATE":
                state["waitlist"] = event["data"]

            if event["type"] == "TABLE_UPDATE":

            # run system logic
            await handle_event(event, state, ws_manager)

    except:
        ws_manager.disconnect(ws)
